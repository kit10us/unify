142 inline
143 ColorUnit::ColorUnit(std::string text, Order order, float defaultAlpha)
144 {
145 using namespace unify::String;
146
147 component.r = 0;
148 component.g = 0;
149 component.b = 0;
150 component.a = defaultAlpha;
151
152 if (StringIs(text, "white"))
153 {
154 component.r = 1.0f;
155 component.g = 1.0f;
156 component.b = 1.0f;
157 }
158 else if (StringIs(text, "black"))
159 {
160 // Nothing (use the above defaults).
161 }
162 else if (StringIs(text, "red"))
163 {
164 component.r = 1.0f;
165 }
166 else if (StringIs(text, "green"))
167 {
168 component.g = 1.0f;
169 }
170 else if (StringIs(text, "blue"))
171 {
172 component.b = 1.0f;
173 }
174 else if (StringIs(text, "clear"))
175 {
176 component.a = 0.0f;
177 }
178 else
179 {
180 std::vector< float > split = Split< float >(text, ',');
181 if (split.size() == 1)
182 {
183 split = SplitOnWhitespace< float >(text);
184 }
185
186 if (split.size() == 3)
187 {
188 component.r = split[0];
189 component.g = split[1];
190 component.b = split[2];
191 }
192 else if (split.size() == 4)
193 {
194 switch (order)
195 {
196 case RGBA:
197 component.r = split[0];
198 component.g = split[1];
199 component.b = split[2];
200 component.a = split[3];
201 break;
202 case ARGB:
203 component.a = split[0];
204 component.r = split[1];
205 component.g = split[2];
206 component.b = split[3];
207 break;
208 case BGRA:
209 component.b = split[0];
210 component.g = split[1];
211 component.r = split[2];
212 component.a = split[3];
213 break;
214 case ABGR:
215 component.a = split[0];
216 component.b = split[1];
217 component.g = split[2];
218 component.r = split[3];
219 break;
220 }
221 }
222 }
223 }

353 std::string ColorUnit::ToString(ColorUnit::Order order) const
354 {
355 switch (order)
356 {
357 case RGBA:
358 default:
359 return Cast< std::string >(component.r) + ", " + Cast< std::string >(component.g) + ", " + Cast< std::string >(component.b) + ", " + Cast< std::string >(component.a);
360 case ARGB:
361 return Cast< std::string >(component.a) + ", " + Cast< std::string >(component.r) + ", " + Cast< std::string >(component.g) + ", " + Cast< std::string >(component.b);
362 case BGRA:
363 return Cast< std::string >(component.b) + ", " + Cast< std::string >(component.g) + ", " + Cast< std::string >(component.r) + ", " + Cast< std::string >(component.a);
364 case ABGR:
365 return Cast< std::string >(component.a) + ", " + Cast< std::string >(component.b) + ", " + Cast< std::string >(component.g) + ", " + Cast< std::string >(component.r);
