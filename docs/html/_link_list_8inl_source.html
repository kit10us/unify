91template< typename T>
92T * LinkList< T >::AddItem( T * item, unsigned int uFlags, float fSortValue, std::string sName )
93{
94 TLink< T > * pNewLink = new TLink< T >( item, sName );
95 if( ! pNewLink ) return 0;
96
97 pNewLink->m_uFlags = uFlags;
98 pNewLink->m_fSortValue = fSortValue;
99
100 if( ! m_pFirst )
101 {
102 m_pFirst = pNewLink;
103 m_pLast = pNewLink;
104 }
105 else
106 {
107 if( unify::CheckFlag(m_uLisFlags, LINKLIST_SORT) )
108 {
109 // Sort into list..
110 TLink< T > * pLink = m_pFirst;
111
112 if( unify::CheckFlag(m_uLisFlags, LINKLIST_SORTREVERSE) )
113 {
114 while( pLink && pLink->m_fSortValue > pNewLink->m_fSortValue )
115 {
116 pLink = pLink->Next();
117 }
118 }
119 else
120 {
121 while( pLink && pLink->m_fSortValue < pNewLink->m_fSortValue )
122 {
123 pLink = pLink->Next();
124 }
125 }
126
127 // Found a larger link
128 if( pLink )
129 {
130 if( pLink == m_pFirst )
131 {
132 m_pFirst = pNewLink;
133 }
134
135 pNewLink->m_previous = pLink->m_previous;
136 if( pNewLink->m_previous )
137 {
138 pNewLink->m_previous->m_next = pNewLink;
139 }
140 pNewLink->m_next = pLink;
141 pLink->m_previous = pNewLink;
142 }
143 else
144 {
145 m_pLast->m_next = pNewLink;
146 pNewLink->m_previous = m_pLast;
147 m_pLast = pNewLink;
148 }
149 }
150
151 else // Non-sorted...
152 {
153 m_pLast->m_next = pNewLink;
154 pNewLink->m_previous = m_pLast;
155 m_pLast = pNewLink;
156 }
157 }
158
159 m_uCount++;
160
161 return item;
162}

164template< typename T >
165bool LinkList< T >::DeleteItem( TLink< T > * & pLink )
166{
168 // REMOVE LINK FROM THE PRIMARY LIST
170
171 // Check if we did not find it...
172 if( ! pLink ) return false;
173
174 // Check if is last...
175 if( m_pLast == pLink ) m_pLast = m_pLast->m_pPrevious;
176 // if is first link...
177 if( m_pFirst == pLink ) m_pFirst = m_pFirst->m_pNext;
178
179
181 // REMOVE LINK FROM ITEM LIST
183
184 if( pLink->m_pPrevious ) pLink->m_pPrevious->m_pNext = pLink->m_pNext;
185 if( pLink->m_pNext ) pLink->m_pNext->m_pPrevious = pLink->m_pPrevious;
186
187 delete pLink;
188
189 m_uCount--;
190
191 return true;
192}
