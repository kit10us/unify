60 void Mark()
61 {
62 // Ensure we are not trying to sample or rate at zero, which would cause a divide by zero error.
63 if (T_SampleSize == 0 || T_SampleRate == 0)
64 {
65 return;
66 }
67
68 auto current_mark = std::chrono::high_resolution_clock::now();
69
70 const auto marked = m_last_mark.has_value();
71
72 // If we have not marked before, we cannot compute a delta, so just return.
73 if (!m_last_mark)
74 {
75 m_last_mark = current_mark;
76 return;
77 }
78
79 std::chrono::duration<float> delta = current_mark - *m_last_mark;
80 m_last_mark = current_mark;
81
82 m_rate_sample += delta;
83
84 m_tick++;
85 if (m_tick >= T_SampleRate)
86 {
87 m_tick = 0;
88
89 auto average = m_rate_sample.count() / T_SampleRate;
90 m_mark[m_next_index++] = average;
91 m_rate_sample = {};
92
93 if (m_next_index >= T_SampleSize)
94 {
95 m_next_index = 0;
96 }
97
98 if (m_samples != std::numeric_limits<size_t>::max())
99 {
100 m_samples++;
101 }
102 }
103 }

113 RateCapture Capture()
114 {
115 if (T_SampleSize == 0 || T_SampleRate == 0 || m_samples == 0)
116 {
117 return {};
118 }
119
120 // Reset our last mark time so we do not have a large delta when we mark again after computing the rate.
121 m_last_mark.reset();
122
123 const size_t count = std::min(m_samples, T_SampleSize);
124 std::copy(m_mark.begin(), m_mark.begin() + count, m_scratch.begin());
125 auto mid_itr = m_scratch.begin() + count / 2;
126 std::nth_element(m_scratch.begin(), mid_itr, m_scratch.begin() + count);
127 const float sum = std::accumulate(m_mark.begin(), m_mark.begin() + count, 0.0f);
128 float median = *mid_itr;
129
130 if (count % 2 == 0)
131 {
132 auto left_itr = std::max_element(m_scratch.begin(), mid_itr);
133 median = (*left_itr + *mid_itr) * 0.5f;
134 }
135
136 RateCapture capture;
137 capture.average = sum / count;
138 capture.median = median;
139 capture.samples = m_samples;
140 capture.sample_size = T_SampleSize;
141 capture.rate = T_SampleRate;
142 return capture;
143 }
