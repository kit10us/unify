40 inline
41 void FileStream::Open(StreamAccessType access, unify::Path path)
42 {
43 Close();
44
45 char mode[3]{};
46 switch (access)
47 {
48 case STREAMACCESS_READBINARY:
49 mode[0] = 'r';
50 mode[1] = 'b';
51 mode[2] = 0;
52 break;
53 case STREAMACCESS_WRITEBINARY:
54 mode[0] = 'w';
55 mode[1] = 'b';
56 mode[2] = 0;
57 break;
58 case STREAMACCESS_APPENDBINARY:
59 mode[0] = 'a';
60 mode[1] = 'b';
61 mode[2] = 0;
62 break;
63 case STREAMACCESS_READ:
64 case STREAMACCESS_READTEXT:
65 mode[0] = 'r';
66 mode[1] = 't';
67 mode[2] = 0;
68 break;
69 case STREAMACCESS_WRITE:
70 case STREAMACCESS_WRITETEXT:
71 mode[0] = 'w';
72 mode[1] = 't';
73 mode[2] = 0;
74 break;
75 case STREAMACCESS_APPEND:
76 case STREAMACCESS_APPENDTEXT:
77 mode[0] = 'a';
78 mode[1] = 't';
79 mode[2] = 0;
80 break;
81
82 case STREAMACCESS_NONE:
83 throw Exception("Attempted to access an FStream with access modifier \'none\'!");
84 };
85
86 m_pStream = fopen(path.ToString().c_str(), mode);
87
88 if (!m_pStream)
89 {
90 throw unify::Exception("Failed to open file \"" + path.ToString() + "\"!");
91 }
92
93 m_Access = access;
94 m_bEndOfStream = false;
95 m_uBytesWaiting = /*BYTES_UNKNOWN*/0;
96 }

104 // Read uLength bytes from stream, return actual bytes read
105 inline
106 unsigned int FileStream::Read(void* pDest, unsigned int uLength)
107 {
108 if (m_Access != STREAMACCESS_READ && m_Access != STREAMACCESS_READBINARY)
109 {
110 return 0;
111 }
112
113 unsigned int u;
114 int c = EOF;
115 for (u = 0; u < uLength; u++)
116 {
117 c = fgetc((FILE*)m_pStream);
118 if (c == EOF) break;
119
120 ((unsigned char*)pDest)[u] = (unsigned char)c;
121 }
122 if (feof((FILE*)m_pStream) || c == EOF)
123 {
124 m_bEndOfStream = true;
125 }
126 return u;
127 }

130 unsigned int FileStream::ReadPack(void* pDest, unsigned int uMaxLength)
131 {
132 if (m_Access != STREAMACCESS_READ && m_Access != STREAMACCESS_READTEXT)
133 {
134 return 0;
135 }
136
137 unsigned int u;
138 int c = 0;
139 c = fgetc((FILE*)m_pStream);
140 for (u = 0; (u < uMaxLength) && c != EOF && c != '\n'; u++)
141 {
142 ((char*)pDest)[u] = (char)c;
143 c = fgetc((FILE*)m_pStream);
144 }
145 ((char*)pDest)[u] = 0;
146 if (feof((FILE*)m_pStream) || c == EOF)
147 {
148 m_bEndOfStream = true;
149 }
150 return u;
151 }

191 {
192 // Reset the end of file...
193 m_bEndOfStream = false;
194
195 switch (seek)
196 {
197 default:
198 return false;
199
200 case STREAMSEEK_START:

202
203 case STREAMSEEK_END:

205
206 case STREAMSEEK_CURRENT:
207 return(fseek((FILE*)m_pStream, iOffset, SEEK_CUR) == 0 ? true : false);
