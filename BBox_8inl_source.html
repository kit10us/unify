185template< typename T >
186bool BBox< T >::Intersects( const Ray & ray ) const
187{
188 V3< float > d = ( ray.direction + 6 - ray.origin ) * 0.5f;
189 V3< float > e = (sup - inf ) * 0.5f;
190 V3< float > c = ray.origin + d - ( inf + sup ) * 0.5f;
191 V3< float > ad = d.Absolute();
192
193 const float EPSILON = std::numeric_limits< float >::epsilon();
194
195 if( fabsf( c[0] ) > e[0] + ad[0] ) return false;
196 if( fabsf( c[1] ) > e[1] + ad[1] ) return false;
197 if( fabsf( c[2] ) > e[2] + ad[2] ) return false;
198
199 if( fabsf( d[1] * c[2] - d[2] * c[1] ) > e[1] * ad[2] + e[2] * ad[1] + EPSILON ) return false;
200 if( fabsf( d[2] * c[0] - d[0] * c[2] ) > e[2] * ad[0] + e[0] * ad[2] + EPSILON ) return false;
201 if( fabsf( d[0] * c[1] - d[1] * c[0] ) > e[0] * ad[1] + e[1] * ad[0] + EPSILON ) return false;
202
203 return true;
204}

206template< typename T >
207bool BBox< T >::Intersects( const Ray & ray, float distanceBegin, float distanceEnd ) const
208{
209 float tmin, tmax, tymin, tymax, tzmin, tzmax;
210 V3< float > invDirection = ray.InvDirection();
211 V3< int > raySign = ray.Sign();
212
213 tmin = ( (raySign.x == 0 ? inf.x : sup.x) - ray.origin.x) * invDirection.x;
214 tmax = ( (raySign.x != 0 ? inf.x : sup.x) - ray.origin.x) * invDirection.x;
215 tymin = ( (raySign.y == 0 ? inf.y : sup.y) - ray.origin.y) * invDirection.y;
216 tymax = ( (raySign.y != 0 ? inf.y : sup.y) - ray.origin.y) * invDirection.y;
217 if( (tmin > tymax) || (tymin > tmax) )
218 {
219 return false;
220 }
221 if( tymin > tmin )
222 {
223 tmin = tymin;
224 }
225 if( tymax < tmax )
226 {
227 tmax = tymax;
228 }
229 tzmin = ( (raySign.z == 0 ? inf.z : sup.z) - ray.origin.z) * invDirection.z;
230 tzmax = ( (raySign.z != 0 ? inf.z : sup.z) - ray.origin.z) * invDirection.z;
231 if( (tmin > tzmax) || (tzmin > tmax) )
232 {
233 return false;
234 }
235 if( tzmin > tmin )
236 {
237 tmin = tzmin;
238 }
239 if( tzmax < tmax )
240 {
241 tmax = tzmax;
242 }
243 return ((tmin < distanceEnd ) && (tmax > distanceBegin ));
244}

246template< typename T >
247bool BBox< T >::Intersects( const Ray & ray, float & distance ) const
248{
249 float tmin, tmax, tymin, tymax, tzmin, tzmax;
250
251 auto orig = ray.origin;
252 auto sign = ray.Sign();
253 auto invdir = ray.InvDirection();
254
255 unify::V3< float > bounds[] = { inf, sup };
256
257 tmin = ( bounds[sign.x].x - orig.x ) * invdir.x;
258 tmax = ( bounds[1 - sign.x].x - orig.x ) * invdir.x;
259 tymin = ( bounds[sign.y].y - orig.y ) * invdir.y;
260 tymax = ( bounds[1 - sign.y].y - orig.y ) * invdir.y;
261
262 if( ( tmin > tymax ) || ( tymin > tmax ) )
263 {
264 return false;
265 }
266
267 if( tymin > tmin )
268 {
269 tmin = tymin;
270 }
271 if( tymax < tmax )
272 {
273 tmax = tymax;
274 }
275
276 tzmin = ( bounds[sign.z].z - orig.z ) * invdir.z;
277 tzmax = ( bounds[1 - sign.y].z - orig.z ) * invdir.z;
278
279 if( ( tmin > tzmax ) || ( tzmin > tmax ) )
280 {
281 return false;
282 }
283
284 if( tzmin > tmin )
285 {
286 tmin = tzmin;
287 }
288 if( tzmax < tmax )
289 {
290 tmax = tzmax;
291 }
292
293 float t = tmin;
294
295 if( t < 0 )
296 {
297 t = tmax;
298 if( t < 0 )
299 {
300 return false;
301 }
302 }
303
304 return true;
305}
