68 Angle::Angle(const std::string_view angle)
69 : m_radians{}
70 {
71 if (angle.length() < 2)
72 {
73 return;
74 }
75
76 const auto start = angle.data();
77 const auto end = start + (angle.length() - 1);
78 auto last = angle.back();
79
80 float value{};
81 auto [ptr, ec] = std::from_chars(start, end, value);
82
83 if (ec == std::errc())
84 {
85 if (last == 'd' || last == 'D')
86 {
87 m_radians = value * DTOR;
88 }
89 else if (last == 'r' || last == 'R')
90 {
91 m_radians = value;
92 }
93 }
94 }

248 int Angle::Normalize()
249 {
250 int magnitude = 0;
251 while (m_radians > TAU)
252 {
253 m_radians -= TAU;
254 magnitude -= 1;
255 }
256 while (m_radians < 0)
257 {
258 m_radians += TAU;
259 magnitude += 1;
260 }
261 return magnitude;
262 }
