41 inline
42 size_t FrameSet::Add(Matrix model, size_t parent, std::string name)
43 {
44 // Assert valid connectivity.
45 size_t index = m_parentIndex.size();
46 assert(parent <= index);
47
48 m_models.push_back(model);
49 m_parentIndex.push_back(parent);
50 if (!name.empty())
51 {
52 m_nameToIndex[name] = index;
53 m_indexToName[index] = name;
54 }
55
56 // Build local matrix.
57 if (parent == index) // No parent.
58 {
59 m_locals.push_back(model);
60 }
61 else
62 {
63 m_locals.push_back(m_locals[parent] * model);
64 }
65
66 return index;
67 }

123 inline
124 void FrameSet::ForEach(ForEachFunctor& functor) const
125 {
126 std::stack< size_t > depth;
127 for (size_t index = 0, end = m_parentIndex.size(); index != end; ++index)
128 {
129 // If our parent isn't immediately before us, then we need to regress.
130 if (!depth.empty() && m_parentIndex[index] != index - 1)
131 {
132 // Reduce our depth until our parent is on the top of the stack or we hit the bottom.
133 while (!depth.empty() && depth.top() != m_parentIndex[index])
134 {
135 depth.pop();
136 }
137 }
138
139 functor(depth.size(), index);
140 depth.push(index);
141 }
142 }
