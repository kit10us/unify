23template< typename T >
24T Parameters::Get( std::string name ) const
25{
26 ParameterMap::const_iterator itr = m_parameters.find( name );
27 if ( itr == m_parameters.end() )
28 {
29 throw Exception( "Failed to find Parameters: \"" + name + "\"!" );
30 }
31 m_auditItemsUsed.insert( name );
32
33 T value {};
34 try
35 {
36 value = unify::any_cast< T >(itr->second.value);
37 }
38 catch( ... )
39 {
40 throw unify::BadCast( std::string( "Failed to cast \"" ) + name + "\" of type " + itr->second.value.type().name() + "!" );
41 }
42
43 return value;
44}
