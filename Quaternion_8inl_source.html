100 Quaternion& Quaternion::operator *= (const Quaternion& quaternion)
101 {
102 Quaternion o;
103 const Quaternion& a = *this;
104 const Quaternion& b = quaternion;
105 o.x = b.w * a.x + b.x * a.w + b.y * a.z - b.z * a.y;
106 o.y = b.w * a.y - b.x * a.z + b.y * a.w + b.z * a.x;
107 o.z = b.w * a.z + b.x * a.y - b.y * a.x + b.z * a.w;
108 o.w = b.w * a.w - b.x * a.x - b.y * a.y - b.z * a.z;
109 *this = o;
110
111 return *this;
112 }

199 inline
200 Quaternion unify::QuaternionFromEuler(const unify::V3< float >& euler)
201 {
202 Quaternion q;
203
204 float c1 = cos(euler.y);
205 float c2 = cos(euler.x);
206 float c3 = cos(euler.z);
207 float s1 = sin(euler.y);
208 float s2 = sin(euler.x);
209 float s3 = sin(euler.z);
210
211 q.w = sqrt(1.0f + c1 * c2 + c1 * c3 - s1 * s2 * s3 + c2 * c3) / 2.0f;
212 q.x = (c2 * s3 + c1 * s3 + s1 * s2 * c3) / (4.0f * q.w);
213 q.y = (s1 * c2 + s1 * c3 + c1 * s2 * s3) / (4.0f * q.w);
214 q.z = (-s1 * s3 + c1 * s2 * c3 + s2) / (4.0f * q.w);
215
216 return q;
217 }

219 inline
220 Quaternion unify::QuaternionSlerp(const Quaternion& qa, const Quaternion& qb, float t)
221 {
222 Quaternion q;
223
224 float cosHalfTheta = qa.w * qb.w + qa.x * qb.x + qa.y * qb.y + qa.z * qb.z;
225 if (fabs(cosHalfTheta) >= 1.0f)
226 {
227 q.w = qa.w; q.x = qa.x; q.y = qa.y; q.z = qa.z;
228 return q;
229 }
230
231 float halfTheta = acos(cosHalfTheta);
232 float sinHalfTheta = sqrt(1.0f - cosHalfTheta * cosHalfTheta);
233
234 if (fabs(sinHalfTheta) < 0.001f)
235 { // fabs is floating point absolute
236 q.w = (qa.w * 0.5f + qb.w * 0.5f);
237 q.x = (qa.x * 0.5f + qb.x * 0.5f);
238 q.y = (qa.y * 0.5f + qb.y * 0.5f);
239 q.z = (qa.z * 0.5f + qb.z * 0.5f);
240 return q;
241 }
242 float ratioA = sin((1.0f - t) * halfTheta) / sinHalfTheta;
243 float ratioB = sin(t * halfTheta) / sinHalfTheta;
244
245 q.w = (qa.w * ratioA + qb.w * ratioB);
246 q.x = (qa.x * ratioA + qb.x * ratioB);
247 q.y = (qa.y * ratioA + qb.y * ratioB);
248 q.z = (qa.z * ratioA + qb.z * ratioB);
249
250 return q;
251 }
