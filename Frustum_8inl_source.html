26 bool unify::EdgeIntersectsFace(unify::V3< float >* pEdges, unify::V3< float >* pFacePoints, unify::Plane* pPlane)
27 {
28 // If both edge points are on the same side of the plane, the edge does
29 // not intersect the face
30 float fDist1;
31 float fDist2;
32 fDist1 = pPlane->a * pEdges[0].x + pPlane->b * pEdges[0].y +
33 pPlane->c * pEdges[0].z + pPlane->d;
34 fDist2 = pPlane->a * pEdges[1].x + pPlane->b * pEdges[1].y +
35 pPlane->c * pEdges[1].z + pPlane->d;
36 if ((fDist1 > 0 && fDist2 > 0) ||
37 (fDist1 < 0 && fDist2 < 0))
38 {
39 return false;
40 }
41
42 // Find point of intersection between edge and face plane (if they're
43 // parallel, edge does not intersect face and D3DXPlaneIntersectLine
44 // returns NULL)
45 unify::V3< float > ptIntersection;
46 if (!pPlane->IntersectLine(ptIntersection, pEdges[0], pEdges[1]))
47 {
48 return false;
49 }
50
51 // Project onto a 2D plane to make the pt-in-poly test easier
52 float fAbsA = (pPlane->a > 0 ? pPlane->a : -pPlane->a);
53 float fAbsB = (pPlane->b > 0 ? pPlane->b : -pPlane->b);
54 float fAbsC = (pPlane->c > 0 ? pPlane->c : -pPlane->c);
55 unify::V2< float > facePoints[4];
56 unify::V2< float > point;
57 if (fAbsA > fAbsB && fAbsA > fAbsC)
58 {
59 // Plane is mainly pointing along X axis, so use Y and Z
60 for (int i = 0; i < 4; i++)
61 {
62 facePoints[i].x = pFacePoints[i].y;
63 facePoints[i].y = pFacePoints[i].z;
64 }
65 point.x = ptIntersection.y;
66 point.y = ptIntersection.z;
67 }
68 else if (fAbsB > fAbsA && fAbsB > fAbsC)
69 {
70 // Plane is mainly pointing along Y axis, so use X and Z
71 for (int i = 0; i < 4; i++)
72 {
73 facePoints[i].x = pFacePoints[i].x;
74 facePoints[i].y = pFacePoints[i].z;
75 }
76 point.x = ptIntersection.x;
77 point.y = ptIntersection.z;
78 }
79 else
80 {
81 // Plane is mainly pointing along Z axis, so use X and Y
82 for (int i = 0; i < 4; i++)
83 {
84 facePoints[i].x = pFacePoints[i].x;
85 facePoints[i].y = pFacePoints[i].y;
86 }
87 point.x = ptIntersection.x;
88 point.y = ptIntersection.y;
89 }
90
91 // If point is on the outside of any of the face edges, it is
92 // outside the face.
93 // We can do this by taking the determinant of the following matrix:
94 // | x0 y0 1 |
95 // | x1 y1 1 |
96 // | x2 y2 1 |
97 // where (x0,y0) and (x1,y1) are points on the face edge and (x2,y2)
98 // is our test point. If this value is positive, the test point is
99 // "to the left" of the line. To determine whether a point needs to
100 // be "to the right" or "to the left" of the four lines to qualify as
101 // inside the face, we need to see if the faces are specified in
102 // clockwise or counter-clockwise order (it could be either, since the
103 // edge could be penetrating from either side). To determine this, we
104 // do the same test to see if the third point is "to the right" or
105 // "to the left" of the line formed by the first two points.
106 // See http://forum.swarthmore.edu/dr.math/problems/scott5.31.96.html
107 float x0, x1, x2, y0, y1, y2;
108 x0 = facePoints[0].x;
109 y0 = facePoints[0].y;
110 x1 = facePoints[1].x;
111 y1 = facePoints[1].y;
112 x2 = facePoints[2].x;
113 y2 = facePoints[2].y;
114 bool bClockwise = false;
115 if (x1 * y2 - y1 * x2 - x0 * y2 + y0 * x2 + x0 * y1 - y0 * x1 < 0)
116 {
117 bClockwise = true;
118 }
119 x2 = point.x;
120 y2 = point.y;
121 for (int i = 0; i < 4; i++)
122 {
123 x0 = facePoints[i].x;
124 y0 = facePoints[i].y;
125
126 if (i < 3)
127 {
128 x1 = facePoints[i + 1].x;
129 y1 = facePoints[i + 1].y;
130 }
131 else
132 {
133 x1 = facePoints[0].x;
134 y1 = facePoints[0].y;
135 }
136
137 if ((x1 * y2 - y1 * x2 - x0 * y2 + y0 * x2 + x0 * y1 - y0 * x1 > 0) == bClockwise)
138 {
139 return false;
140 }
141 }
142
143 // If we get here, the point is inside all four face edges,
144 // so it's inside the face.
145 return true;
146 }

160 void Frustum::Calculate(const unify::Matrix& worldViewProjection)
161 {
162 m_worldViewProjection = worldViewProjection;
163 unify::Matrix result = worldViewProjection;
164 result.Invert();
165
166 m_vCorner[FrustumCorner::xyz] = unify::V3< float >(-1.0f, -1.0f, 0.0f);
167 m_vCorner[FrustumCorner::Xyz] = unify::V3< float >(1.0f, -1.0f, 0.0f);
168 m_vCorner[FrustumCorner::xYz] = unify::V3< float >(-1.0f, 1.0f, 0.0f);
169 m_vCorner[FrustumCorner::XYz] = unify::V3< float >(1.0f, 1.0f, 0.0f);
170 m_vCorner[FrustumCorner::xyZ] = unify::V3< float >(-1.0f, -1.0f, 1.0f);
171 m_vCorner[FrustumCorner::XyZ] = unify::V3< float >(1.0f, -1.0f, 1.0f);
172 m_vCorner[FrustumCorner::xYZ] = unify::V3< float >(-1.0f, 1.0f, 1.0f);
173 m_vCorner[FrustumCorner::XYZ] = unify::V3< float >(1.0f, 1.0f, 1.0f);
174
175 int i;
176 for (i = 0; i < 8; i++)
177 {
178 m_vCorner[i] = result.TransformCoord(m_vCorner[i]);
179 }
180
181 m_Plane[FrustumPlane::Near] = unify::Plane::PlaneFromPoints(m_vCorner[FrustumCorner::xyz], m_vCorner[FrustumCorner::Xyz], m_vCorner[FrustumCorner::xYz]);
182 m_Plane[FrustumPlane::Far] = unify::Plane::PlaneFromPoints(m_vCorner[FrustumCorner::xYZ], m_vCorner[FrustumCorner::XYZ], m_vCorner[FrustumCorner::XyZ]);
183 m_Plane[FrustumPlane::Left] = unify::Plane::PlaneFromPoints(m_vCorner[FrustumCorner::xYz], m_vCorner[FrustumCorner::xYZ], m_vCorner[FrustumCorner::xyZ]);
184 m_Plane[FrustumPlane::Right] = unify::Plane::PlaneFromPoints(m_vCorner[FrustumCorner::XYZ], m_vCorner[FrustumCorner::XYz], m_vCorner[FrustumCorner::XyZ]);
185 m_Plane[FrustumPlane::Top] = unify::Plane::PlaneFromPoints(m_vCorner[FrustumCorner::xYz], m_vCorner[FrustumCorner::XYz], m_vCorner[FrustumCorner::xYZ]);
186 m_Plane[FrustumPlane::Bottom] = unify::Plane::PlaneFromPoints(m_vCorner[FrustumCorner::Xyz], m_vCorner[FrustumCorner::xyz], m_vCorner[FrustumCorner::xyZ]);
187 }

189 inline
190 bool Frustum::CastPoint(const unify::V2< float >& unit, unify::Ray& rayOut)
191 {
192 if (unit.x < 0 || unit.y < 0 || unit.x > 1.0f || unit.y > 1.0f)
193 {
194 return false;
195 }
196
197 unify::V3< float > vNear1 = unify::V3< float >::V3Lerp(m_vCorner[FrustumCorner::xYz], m_vCorner[FrustumCorner::XYz], unit.x);
198 unify::V3< float > vNear2 = unify::V3< float >::V3Lerp(m_vCorner[FrustumCorner::xyz], m_vCorner[FrustumCorner::Xyz], unit.x);
199 unify::V3< float > vNear = unify::V3< float >::V3Lerp(vNear1, vNear2, unit.y);
200
201 unify::V3< float > vFar1 = unify::V3< float >::V3Lerp(m_vCorner[FrustumCorner::xYZ], m_vCorner[FrustumCorner::XYZ], unit.x);
202 unify::V3< float > vFar2 = unify::V3< float >::V3Lerp(m_vCorner[FrustumCorner::xyZ], m_vCorner[FrustumCorner::XyZ], unit.x);
203 unify::V3< float > vFar = unify::V3< float >::V3Lerp(vFar1, vFar2, unit.y);
204
205 rayOut.origin = vNear;
206 rayOut.direction = vFar - vNear;
207 rayOut.direction.Normalize();
208
209 return true;
210 }

213 CullState Frustum::CullBBox(unify::BBox< float >* pBBox/*, unify::Plane * BBPlane*/)
214 {
215 unsigned char outside[8]{};
216
217 unify::V3< float > vBounds[8];
218 pBBox->GenerateCorners(&vBounds[0]);
219
220 unify::Plane BBPlane[6];
221
222 BBPlane[0] = unify::Plane::PlaneFromPoints(vBounds[0], vBounds[1], vBounds[2]); // Near
223 BBPlane[1] = unify::Plane::PlaneFromPoints(vBounds[6], vBounds[7], vBounds[5]); // Far
224 BBPlane[2] = unify::Plane::PlaneFromPoints(vBounds[2], vBounds[6], vBounds[4]); // Left
225 BBPlane[3] = unify::Plane::PlaneFromPoints(vBounds[7], vBounds[3], vBounds[5]); // Right
226 BBPlane[4] = unify::Plane::PlaneFromPoints(vBounds[2], vBounds[3], vBounds[6]); // Top
227 BBPlane[5] = unify::Plane::PlaneFromPoints(vBounds[1], vBounds[0], vBounds[4]); // Bottom
228
229 // Check boundary vertices against all 6 frustum planes,
230 // and store result (1 if outside) in a bitfield
231 for (int iPoint = 0; iPoint < 8; iPoint++)
232 {
233 for (int iPlane = 0; iPlane < 6; iPlane++)
234 {
235 if (m_Plane[iPlane].a * vBounds[iPoint].x +
236 m_Plane[iPlane].b * vBounds[iPoint].y +
237 m_Plane[iPlane].c * vBounds[iPoint].z +
238 m_Plane[iPlane].d < 0)
239 {
240 // TODO: Figure this out.
241 outside[iPoint] |= (1 << iPlane);
242 }
243 }
244 // If any point is inside all 6 frustum planes, it is inside
245 // the frustum, so the object must be rendered.
246 if (outside[iPoint] == 0)
247 {
248 return CullState::Inside;
249 }
250 }
251
252 // If all points are outside any single frustum plane, the object is
253 // outside the frustum
254 if ((outside[0] & outside[1] & outside[2] & outside[3] & outside[4] & outside[5] & outside[6] & outside[7]) != 0)
255 {
256 return CullState::Outside;
257 }
258
259 // Now see if any of the frustum edges penetrate any of the faces of
260 // the bounding box
261 unify::V3< float > edge[12][2] =
262 {
263 m_vCorner[0], m_vCorner[1], // front bottom
264 m_vCorner[2], m_vCorner[3], // front top
265 m_vCorner[0], m_vCorner[2], // front left
266 m_vCorner[1], m_vCorner[3], // front right
267 m_vCorner[4], m_vCorner[5], // back bottom
268 m_vCorner[6], m_vCorner[7], // back top
269 m_vCorner[4], m_vCorner[6], // back left
270 m_vCorner[5], m_vCorner[7], // back right
271 m_vCorner[0], m_vCorner[4], // left bottom
272 m_vCorner[2], m_vCorner[6], // left top
273 m_vCorner[1], m_vCorner[5], // right bottom
274 m_vCorner[3], m_vCorner[7], // right top
275 };
276 unify::V3< float > face[6][4] =
277 {
278 vBounds[0], vBounds[2], vBounds[3], vBounds[1], // front
279 vBounds[4], vBounds[5], vBounds[7], vBounds[6], // back
280 vBounds[0], vBounds[4], vBounds[6], vBounds[2], // left
281 vBounds[1], vBounds[3], vBounds[7], vBounds[5], // right
282 vBounds[2], vBounds[6], vBounds[7], vBounds[3], // top
283 vBounds[0], vBounds[4], vBounds[5], vBounds[1], // bottom
284 };
285 unify::V3< float >* pEdge;
286 unify::V3< float >* pFace;
287 pEdge = &edge[0][0];
288 for (int iEdge = 0; iEdge < 12; iEdge++)
289 {
290 pFace = &face[0][0];
291 for (int iFace = 0; iFace < 6; iFace++)
292 {
293 if (EdgeIntersectsFace(pEdge, pFace, &BBPlane[iFace]))
294 {
295 return CullState::InsideSlow;
296 }
297 pFace += 4;
298 }
299 pEdge += 2;
300 }
301
302 // Now see if any of the bounding box edges penetrate any of the faces of
303 // the frustum
304 unify::V3< float > edge2[12][2] =
305 {
306 vBounds[0], vBounds[1], // front bottom
307 vBounds[2], vBounds[3], // front top
308 vBounds[0], vBounds[2], // front left
309 vBounds[1], vBounds[3], // front right
310 vBounds[4], vBounds[5], // back bottom
311 vBounds[6], vBounds[7], // back top
312 vBounds[4], vBounds[6], // back left
313 vBounds[5], vBounds[7], // back right
314 vBounds[0], vBounds[4], // left bottom
315 vBounds[2], vBounds[6], // left top
316 vBounds[1], vBounds[5], // right bottom
317 vBounds[3], vBounds[7], // right top
318 };
319 unify::V3< float > face2[6][4] =
320 {
321 m_vCorner[0], m_vCorner[2], m_vCorner[3], m_vCorner[1], // front
322 m_vCorner[4], m_vCorner[5], m_vCorner[7], m_vCorner[6], // back
323 m_vCorner[0], m_vCorner[4], m_vCorner[6], m_vCorner[2], // left
324 m_vCorner[1], m_vCorner[3], m_vCorner[7], m_vCorner[5], // right
325 m_vCorner[2], m_vCorner[6], m_vCorner[7], m_vCorner[3], // top
326 m_vCorner[0], m_vCorner[4], m_vCorner[5], m_vCorner[1], // bottom
327 };
328 pEdge = &edge2[0][0];
329 for (int iEdge = 0; iEdge < 12; iEdge++)
330 {
331 pFace = &face2[0][0];
332 for (int iFace = 0; iFace < 6; iFace++)
333 {
334 if (EdgeIntersectsFace(pEdge, pFace, &m_Plane[iFace]))
335 {
336 return CullState::InsideSlow;
337 }
338 pFace += 4;
339 }
340 pEdge += 2;
341 }
342
343 // Now see if frustum is contained in bounding box
344 // If any frustum corner point is outside any plane of the bounding box,
345 // the frustum is not contained in the bounding box, so the object
346 // is outside the frustum
347 for (int iPlane = 0; iPlane < 6; iPlane++)
348 {
349 if (BBPlane[iPlane].a * m_vCorner[0].x +
350 BBPlane[iPlane].b * m_vCorner[0].y +
351 BBPlane[iPlane].c * m_vCorner[0].z +
352 BBPlane[iPlane].d < 0)
353 {
354 return CullState::OutsideSlow;
355 }
356 }
357
358 // Bounding box must contain the frustum, so render the object
359 return CullState::InsideSlow;
360 }
