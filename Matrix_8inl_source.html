31 inline
32 Matrix::Matrix( Quaternion q, V3< float > translation )
33 {
34 /*
35 const float & x = q.x;
36 const float & y = q.y;
37 const float & z = q.z;
38 const float & w = q.w;
39 float x2 = x * x;
40 float y2 = y * y;
41 float z2 = z * z;
42
43 float xy = x * y;
44 float wy = w * y;
45 float xz = x * z;
46 float yz = y * z;
47 float wx = w * x;
48 float wz = w * z;
49
50 Matrix rotation;
51 rotation.m[0][0] = 1.0f - 2.0f * y2 - 2.0f * z2;
52 rotation.m[1][0] = 2.0f * xy - 2.0f * wz;
53 rotation.m[2][0] = 2.0f * xz + 2.0f * wy;
54 rotation.m[3][0] = 0.0f;
55
56 rotation.m[0][1] = 2.0f * xy + 2.0f * wz;
57 rotation.m[1][1] = 1.0f - 2.0f * x2 - 2.0f * z2;
58 rotation.m[2][1] = 2.0f * yz - 2.0f * wx;
59 rotation.m[3][1] = 0.0f;
60
61 rotation.m[0][2] = 2.0f * xz - 2.0f * wy;
62 rotation.m[1][2] = 2.0f * yz + 2.0f * wx;
63 rotation.m[2][2] = 1.0f - 2.0f * x2 - 2.0f * y2;
64 rotation.m[3][2] = 0.0f;
65
66 rotation.m[0][3] = 0.0f;
67 rotation.m[1][3] = 0.0f;
68 rotation.m[2][3] = 0.0f;
69 rotation.m[3][3] = 1.0f;
70 */
71
72 /*
73 Matrix scaleM( MatrixScale( scale ) );
74 Matrix translationM( MatrixTranslate( translation ) );
75 *this = scaleM * rotation * translationM;
76 */
77
78 /*
79
80 // invs (inverse square length) is only required if quaternion is not already normalised
81 float invs = 1 / (sqx + sqy + sqz + sqw);
82 m[0][0] = (sqx - sqy - sqz + sqw)*invs; // since sqw + sqx + sqy + sqz =1/invs*invs
83 m[1][1] = (-sqx + sqy - sqz + sqw)*invs;
84 m[2][2] = (-sqx - sqy + sqz + sqw)*invs;
85
86 float tmp1 = q.x*q.y;
87 float tmp2 = q.z*q.w;
88 m[0][1] = 2.0f * (tmp1 + tmp2)*invs;
89 m[1][0] = 2.0f * (tmp1 - tmp2)*invs;
90
91 tmp1 = q.x*q.z;
92 tmp2 = q.y*q.w;
93 m[0][2] = 2.0f * (tmp1 - tmp2)*invs;
94 m[2][0] = 2.0f * (tmp1 + tmp2)*invs;
95 tmp1 = q.y*q.z;
96 tmp2 = q.x*q.w;
97 m[1][2] = 2.0f * (tmp1 + tmp2)*invs;
98 m[2][1] = 2.0f * (tmp1 - tmp2)*invs;
99
100 m[3][0] = translation.x;
101 m[3][1] = translation.y;
102 m[3][2] = translation.z;
103
104 m[0][3] = 0.0f;
105 m[1][3] = 0.0f;
106 m[2][3] = 0.0f;
107
108 m[3][3] = 1.0f;
109 */
110
111 //q = q.Conjugate();
112
113 const float & x = q.x;
114 const float & y = q.y;
115 const float & z = q.z;
116 const float & w = q.w;
117
118 float xs = x * x;
119 float ys = y * y;
120 float zs = z * z;
121
122 SetRow( 0, { (1.0f - 2.0f * ys - 2.0f * zs), (2.0f * x * y + 2.0f * w * z), (2 * x * z - 2.0f * w * y), 0.0f } );
123 SetRow( 1, { (2.0f * x * y - 2.0f * w * z), (1.0f - 2.0f * xs - 2.0f * zs), (2.0f * y * z + 2.0f * w * x), 0.0f } );
124 SetRow( 2, { (2.0f * x * z + 2.0f * w * y), (2.0f * y * z - 2.0f * w * x), (1.0f - 2.0f * xs - 2.0f * ys), 0.0f } );
125 SetRow( 3, { translation, 1.0f } );
126 }

587 inline
588 Quaternion Matrix::GetRotation() const
589 {
590 Quaternion q;
591
592 float trace = m[0][0] + m[1][1] + m[2][2]; // I removed + 1.0f; see discussion with Ethan
593 if( trace > 0 ) {// I changed M_EPSILON to 0
594 float s = 0.5f / sqrtf( trace + 1.0f );
595 q.w = 0.25f / s;
596 q.x = (m[2][1] - m[1][2]) * s;
597 q.y = (m[0][2] - m[2][0]) * s;
598 q.z = (m[1][0] - m[0][1]) * s;
599 }
600 else {
601 if( m[0][0] > m[1][1] && m[0][0] > m[2][2] ) {
602 float s = 2.0f * sqrtf( 1.0f + m[0][0] - m[1][1] - m[2][2] );
603 q.w = (m[2][1] - m[1][2]) / s;
604 q.x = 0.25f * s;
605 q.y = (m[0][1] + m[1][0]) / s;
606 q.z = (m[0][2] + m[2][0]) / s;
607 }
608 else if( m[1][1] > m[2][2] ) {
609 float s = 2.0f * sqrtf( 1.0f + m[1][1] - m[0][0] - m[2][2] );
610 q.w = (m[0][2] - m[2][0]) / s;
611 q.x = (m[0][1] + m[1][0]) / s;
612 q.y = 0.25f * s;
613 q.z = (m[1][2] + m[2][1]) / s;
614 }
615 else {
616 float s = 2.0f * sqrtf( 1.0f + m[2][2] - m[0][0] - m[1][1] );
617 q.w = (m[1][0] - m[0][1]) / s;
618 q.x = (m[0][2] + m[2][0]) / s;
619 q.y = (m[1][2] + m[2][1]) / s;
620 q.z = 0.25f * s;
621 }
622 }
623
624 return q;
625 }

636 inline
637 float DeterminantFunc( const float * in, float * out )
638 {
639 float inv[16], det;
640
641 inv[0] = in[5] * in[10] * in[15] -
642 in[5] * in[11] * in[14] -
643 in[9] * in[6] * in[15] +
644 in[9] * in[7] * in[14] +

697 inv[2] = in[1] * in[6] * in[15] -
698 in[1] * in[7] * in[14] -
699 in[5] * in[2] * in[15] +

727 in[5] * in[2] * in[11] -
728 in[5] * in[3] * in[10] -

731
732 inv[7] = in[0] * in[6] * in[11] -

736 in[8] * in[2] * in[7] -
737 in[8] * in[3] * in[6];

749 in[4] * in[2] * in[9] +
750 in[8] * in[1] * in[6] -
751 in[8] * in[2] * in[5];
752
753 det = in[0] * inv[0] + in[1] * inv[4] + in[2] * inv[8] + in[3] * inv[12];
754
755 if( det == 0 )
756 {
757 return 0;
758 }
759
760 det = 1.0f / det;
761
762 if ( out != nullptr )
763 {
764 for( int i = 0; i < 16; i++ )
765 {
766 out[i] = inv[i] * det;
767 }
768 }
769
770 return det;

1232 yaxis = cross( zaxis, xaxis )

1240 inline
1241 Matrix MatrixLookAtLH( const V3< float > & eye, const V3< float > & at, const V3< float > & up )
1242 {
1243 Matrix matrix = unify::MatrixIdentity();
1244
1245 typedef unify::V3< float > V;
1246
1247 V zaxis = V::V3Normalized( at - eye );
1248 V xaxis = V::V3Normalized( V::V3Cross( up, zaxis ) );
1249 V yaxis = V::V3Cross( zaxis, xaxis );
1250
