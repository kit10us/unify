197 inline
198 Quaternion ToQuaternion(const Matrix& mat)
199 {
200 Quaternion q;
201 float tr = mat.m[0][0] + mat.m[1][1] + mat.m[2][2];
202
203 if (tr > 0)
204 {
205 float S = sqrt(tr + 1.0f) * 2.0f; // S=4*qw
206 q.w = 0.25f * S;
207 q.x = (mat.m[1][2] - mat.m[2][1]) / S;
208 q.y = (mat.m[2][0] - mat.m[0][2]) / S;
209 q.z = (mat.m[0][1] - mat.m[1][0]) / S;
210 }
211 else if ((mat.m[0][0] > mat.m[1][1]) & (mat.m[0][0] > mat.m[2][2]))
212 {
213 float S = sqrt(1.0f + mat.m[0][0] - mat.m[1][1] - mat.m[2][2]) * 2.0f; // S=4*qx
214 q.w = (mat.m[1][2] - mat.m[2][1]) / S;
215 q.x = 0.25f * S;
216 q.y = (mat.m[1][0] + mat.m[0][1]) / S;
217 q.z = (mat.m[2][0] + mat.m[0][2]) / S;
218 }
219 else if (mat.m[1][1] > mat.m[2][2])
220 {
221 float S = sqrt(1.0f + mat.m[1][1] - mat.m[0][0] - mat.m[2][2]) * 2.0f; // S=4*qy
222 q.w = (mat.m[2][0] - mat.m[0][2]) / S;
223 q.x = (mat.m[1][0] + mat.m[0][1]) / S;
224 q.y = 0.25f * S;
225 q.z = (mat.m[2][1] + mat.m[1][2]) / S;
226 }
227 else
228 {
229 float S = sqrt(1.0f + mat.m[2][2] - mat.m[0][0] - mat.m[1][1]) * 2.0f; // S=4*qz
230 q.w = (mat.m[0][1] - mat.m[1][0]) / S;
231 q.x = (mat.m[2][0] + mat.m[0][2]) / S;
232 q.y = (mat.m[2][1] + mat.m[1][2]) / S;
233 q.z = 0.25f * S;
234 }
235 return q;
236 }
