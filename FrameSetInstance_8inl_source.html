140 inline
141 void FrameSetInstance::UpdateLocals()
142 {
143 if (m_dirty)
144 {
145 for (size_t i = 0, end = Count(); i != end; ++i)
146 {
147 const size_t& parentIndex = m_frameSet->m_parentIndex[i];
148 if (i == parentIndex)
149 {
150 m_locals[i] = m_models[i];
151 }
152 else
153 {
154 const Matrix& parentLocal = m_locals[parentIndex];
155 m_locals[i] = parentLocal * m_models[i];
156 }
157 }
158 m_dirty = false;
159 }
160 }
